private void copyNonNullProperties(Object source, Object target) { BeanWrapper src = new BeanWrapperImpl(source); PropertyDescriptor[] pds = src.getPropertyDescriptors(); for (PropertyDescriptor pd : pds) { // Exclure la propriété 'class' if (!pd.getName().equals("class")) { Object srcValue = src.getPropertyValue(pd.getName()); if (srcValue != null) { BeanWrapper targetWrapper = new BeanWrapperImpl(target); targetWrapper.setPropertyValue(pd.getName(), srcValue); } } } }
