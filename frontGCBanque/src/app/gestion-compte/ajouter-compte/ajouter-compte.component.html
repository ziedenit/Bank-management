private void merge(Object target, Object source) { Field[] fields = source.getClass().getDeclaredFields(); for (Field field : fields) { field.setAccessible(true); try { Object value = field.get(source); if (value != null) { if (value instanceof String && ((String) value).trim().isEmpty()) { // If the value is a string and it's empty, set the target field to null field.set(target, null); } else if (isComplexObject(field.getType())) { Object targetValue = field.get(target); if (targetValue == null) { field.set(target, value); } else { merge(targetValue, value); } } else if (field.getType().equals(Date.class)) { if (((String) value).trim().isEmpty()) { field.set(target, null); // Set Date field to null if the value is an empty string } else { // Parse the date string and set the Date field SimpleDateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss.SSSXXX"); Date parsedDate = dateFormat.parse((String) value); field.set(target, parsedDate); } } else { field.set(target, value); } } } catch (IllegalAccessException | ParseException e) { throw new RuntimeException("Failed to merge field: " + field.getName(), e); } } }
